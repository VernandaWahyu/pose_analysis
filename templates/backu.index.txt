    function calculateAngle(a, b, c) {
        // a, b, c adalah object {x, y}
        const ba = { x: a.x - b.x, y: a.y - b.y };
        const bc = { x: c.x - b.x, y: c.y - b.y };
        
        const dot = ba.x * bc.x + ba.y * bc.y;
        const mag = Math.sqrt(ba.x * ba.x + ba.y * ba.y) * 
                    Math.sqrt(bc.x * bc.x + bc.y * bc.y);
        
        if (mag === 0) return 0;
        
        let cosAngle = dot / mag;
        cosAngle = Math.max(-1, Math.min(1, cosAngle));
        const angleRad = Math.acos(cosAngle);
        return angleRad * (180 / Math.PI);
    }

    // ==========================================
    // FUNGSI ANALISIS - SAMA DENGAN PYTHON
    // ==========================================
    function analyzePoseData(poseData) {
        if (!poseData || poseData.length < 2) {
            return null;
        }

        // Kumpulkan data dari semua frame
        const leftKneeAngles = [];
        const rightKneeAngles = [];
        const leftElbowAngles = [];
        const rightElbowAngles = [];
        const hipPositions = [];
        const leftAnkleY = [];
        const rightAnkleY = [];
        const leftWristY = [];
        const rightWristY = [];
        const leftHipX = [];
        const rightHipX = [];
        const leftHipY = [];
        const rightHipY = [];

        poseData.forEach((pose) => {
            const landmarks = pose.landmarks;
            if (!landmarks) return;

            // Ambil koordinat (index MediaPipe)
            const leftHip = landmarks[23];
            const rightHip = landmarks[24];
            const leftKnee = landmarks[25];
            const rightKnee = landmarks[26];
            const leftAnkle = landmarks[27];
            const rightAnkle = landmarks[28];
            const leftShoulder = landmarks[11];
            const rightShoulder = landmarks[12];
            const leftElbow = landmarks[13];
            const rightElbow = landmarks[14];
            const leftWrist = landmarks[15];
            const rightWrist = landmarks[16];

            if (!leftHip || !rightHip || !leftKnee || !rightKnee || 
                !leftAnkle || !rightAnkle || !leftShoulder || !rightShoulder ||
                !leftElbow || !rightElbow || !leftWrist || !rightWrist) return;

            // Hitung angle knee
            const leftKneeAngle = calculateAngle(
                { x: leftHip.x, y: leftHip.y },
                { x: leftKnee.x, y: leftKnee.y },
                { x: leftAnkle.x, y: leftAnkle.y }
            );
            const rightKneeAngle = calculateAngle(
                { x: rightHip.x, y: rightHip.y },
                { x: rightKnee.x, y: rightKnee.y },
                { x: rightAnkle.x, y: rightAnkle.y }
            );

            // Hitung angle elbow
            const leftElbowAngle = calculateAngle(
                { x: leftShoulder.x, y: leftShoulder.y },
                { x: leftElbow.x, y: leftElbow.y },
                { x: leftWrist.x, y: leftWrist.y }
            );
            const rightElbowAngle = calculateAngle(
                { x: rightShoulder.x, y: rightShoulder.y },
                { x: rightElbow.x, y: rightElbow.y },
                { x: rightWrist.x, y: rightWrist.y }
            );

            leftKneeAngles.push(leftKneeAngle);
            rightKneeAngles.push(rightKneeAngle);
            leftElbowAngles.push(leftElbowAngle);
            rightElbowAngles.push(rightElbowAngle);

            // Hip center
            const hipX = (leftHip.x + rightHip.x) / 2;
            const hipY = (leftHip.y + rightHip.y) / 2;
            hipPositions.push({ x: hipX, y: hipY });

            // Ankle Y
            leftAnkleY.push(leftAnkle.y);
            rightAnkleY.push(rightAnkle.y);

            // Wrist Y
            leftWristY.push(leftWrist.y);
            rightWristY.push(rightWrist.y);

            // Hip X & Y
            leftHipX.push(leftHip.x);
            rightHipX.push(rightHip.x);
            leftHipY.push(leftHip.y);
            rightHipY.push(rightHip.y);
        });

        // ==========================================
        // BODY SHAPE
        // ==========================================
        const allKneeAngles = [...leftKneeAngles, ...rightKneeAngles];
        const meanKneeAngle = allKneeAngles.length > 0 ? 
            allKneeAngles.reduce((a, b) => a + b, 0) / allKneeAngles.length : 0;

        const allElbowAngles = [...leftElbowAngles, ...rightElbowAngles];
        const meanElbowAngle = allElbowAngles.length > 0 ?
            allElbowAngles.reduce((a, b) => a + b, 0) / allElbowAngles.length : 0;

        // Foot Lift
        const leftAnkleBaseline = Math.max(...leftAnkleY);
        const rightAnkleBaseline = Math.max(...rightAnkleY);

        const leftLifts = leftAnkleY.map(y => Math.max(0, leftAnkleBaseline - y));
        const rightLifts = rightAnkleY.map(y => Math.max(0, rightAnkleBaseline - y));
        const allLifts = [...leftLifts, ...rightLifts];
        const meanFootLift = allLifts.length > 0 ? 
            allLifts.reduce((a, b) => a + b, 0) / allLifts.length : 0;
        const maxFootLift = allLifts.length > 0 ? Math.max(...allLifts) : 0;

        // Arm Lift
        const leftWristBaseline = Math.max(...leftWristY);
        const rightWristBaseline = Math.max(...rightWristY);

        const leftArmLifts = leftWristY.map(y => Math.max(0, leftWristBaseline - y));
        const rightArmLifts = rightWristY.map(y => Math.max(0, rightWristBaseline - y));
        const allArmLifts = [...leftArmLifts, ...rightArmLifts];
        const meanArmLift = allArmLifts.length > 0 ?
            allArmLifts.reduce((a, b) => a + b, 0) / allArmLifts.length : 0;
        const maxArmLift = allArmLifts.length > 0 ? Math.max(...allArmLifts) : 0;

        // ==========================================
        // EFFORT
        // ==========================================
        const velocities = [];
        for (let i = 1; i < hipPositions.length; i++) {
            const dx = hipPositions[i].x - hipPositions[i-1].x;
            const dy = hipPositions[i].y - hipPositions[i-1].y;
            const d = Math.sqrt(dx*dx + dy*dy);
            velocities.push(d);
        }

        const accelerations = [];
        for (let i = 1; i < velocities.length; i++) {
            accelerations.push(Math.abs(velocities[i] - velocities[i-1]));
        }

        const meanVelocity = velocities.length > 0 ?
            velocities.reduce((a, b) => a + b, 0) / velocities.length : 0;
        const meanAcceleration = accelerations.length > 0 ?
            accelerations.reduce((a, b) => a + b, 0) / accelerations.length : 0;
        const motionEnergy = velocities.length > 0 ?
            velocities.reduce((a, b) => a + b * b, 0) / velocities.length : 0;

        // ==========================================
        // SPACE
        // ==========================================
        const meanMovement = meanVelocity;
        const maxMovement = velocities.length > 0 ? Math.max(...velocities) : 0;

        // Direction Consistency
        const directions = [];
        for (let i = 1; i < hipPositions.length; i++) {
            const dx = hipPositions[i].x - hipPositions[i-1].x;
            directions.push(dx);
        }

        let directionChanges = 0;
        const THRESHOLD = 0.03; // Dinormalisasi karena koordinat 0-1

        for (let i = 1; i < directions.length; i++) {
            const prev = directions[i-1];
            const curr = directions[i];
            if (Math.abs(prev) > THRESHOLD && Math.abs(curr) > THRESHOLD) {
                if (Math.sign(prev) !== Math.sign(curr)) {
                    directionChanges++;
                }
            }
        }

        const directionConsistency = directions.length > 0 ?
            1 - (directionChanges / directions.length) : 0;

        // ==========================================
        // KATEGORI
        // ==========================================
        const kneeCat = kategoriMeanKneeAngle(meanKneeAngle);
        const footCat = kategoriMeanFootLift(meanFootLift);
        const elbowCat = kategoriMeanElbowAngle(meanElbowAngle);
        const armCat = kategoriMeanArmLift(meanArmLift);
        const maxArmCat = kategoriMaxArmLift(maxArmLift);
        const velocityCat = kategoriVelocity(meanVelocity);
        const accelerationCat = kategoriAcceleration(meanAcceleration);
        const energyCat = kategoriMotionEnergy(motionEnergy);
        const movementCat = kategoriMeanMovement(meanMovement);
        const maxMovementCat = kategoriMaxMovement(maxMovement);
        const directionCat = kategoriDirectionConsistency(directionConsistency);

        return {
            meanKneeAngle,
            meanFootLift,
            maxFootLift,
            meanElbowAngle,
            meanArmLift,
            maxArmLift,
            meanVelocity,
            meanAcceleration,
            motionEnergy,
            meanMovement,
            maxMovement,
            directionConsistency,
            kneeCat,
            footCat,
            elbowCat,
            armCat,
            maxArmCat,
            velocityCat,
            accelerationCat,
            energyCat,
            movementCat,
            maxMovementCat,
            directionCat
        };
    }

    function drawPose(ctx, landmarks) {
        const connections = [
            [11, 12], [11, 13], [13, 15], [12, 14], [14, 16],
            [23, 24], [23, 25], [25, 27], [24, 26], [26, 28],
            [11, 23], [12, 24]
        ];
        
        ctx.strokeStyle = '#00ff88';
        ctx.lineWidth = 2;
        
        connections.forEach(([i, j]) => {
            const a = landmarks[i];
            const b = landmarks[j];
            if (a && b && a.visibility > 0.5 && b.visibility > 0.5) {
                ctx.beginPath();
                ctx.moveTo(a.x * output.width, a.y * output.height);
                ctx.lineTo(b.x * output.width, b.y * output.height);
                ctx.stroke();
            }
        });
        
        landmarks.forEach(landmark => {
            if (landmark.visibility > 0.5) {
                ctx.beginPath();
                ctx.arc(landmark.x * output.width, landmark.y * output.height, 4, 0, 2 * Math.PI);
                ctx.fillStyle = '#ff0080';
                ctx.fill();
            }
        });
    }

    async function startProcessing() {
        const file = videoInput.files[0];
        if (!file) {
            alert('Silakan upload video terlebih dahulu!');
            return;
        }
        
        console.log('Memulai analisis...');
        
        progressSection.classList.remove('hidden');
        progressBar.style.width = '0%';
        progressText.textContent = '0%';
        statusDiv.innerHTML = '<i class="fas fa-spinner fa-spin mr-1"></i> Memproses video...';
        startBtn.disabled = true;
        
        // Data storage
        const poseData = [];
        let frameCount = 0;
        let isRunning = true;
        
        try {
            // Setup MediaPipe
            const pose = new Pose({
                locateFile: (file) => {
                    return `https://cdn.jsdelivr.net/npm/@mediapipe/pose/${file}`;
                }
            });
            
            pose.setOptions({
                modelComplexity: 1,
                smoothLandmarks: true,
                minDetectionConfidence: 0.5,
                minTrackingConfidence: 0.5
            });
            
            pose.onResults((results) => {
                if (results.poseLandmarks) {
                    poseData.push({
                        landmarks: results.poseLandmarks,
                        timestamp: Date.now()
                    });
                    
                    frameCount++;
                    const progress = Math.min((frameCount / 100) * 100, 100);
                    progressBar.style.width = progress + '%';
                    progressText.textContent = Math.round(progress) + '%';
                    
                    canvasCtx.clearRect(0, 0, output.width, output.height);
                    drawPose(canvasCtx, results.poseLandmarks);
                }
            });
            
            // Load video
            const url = URL.createObjectURL(file);
            video.src = url;
            video.load();
            
            await new Promise((resolve) => {
                video.onloadedmetadata = () => resolve();
            });
            
            await video.play();
            
            output.width = video.videoWidth || 640;
            output.height = video.videoHeight || 480;
            
            statusDiv.innerHTML = '<i class="fas fa-play mr-1"></i> Menganalisis...';
            
            // Process frames dengan interval yang lebih lambat agar tidak overload
            let lastTime = 0;
            const interval = 1000 / 15; // 15 fps untuk performa lebih baik
            
            async function processFrame() {
                if (!isRunning) return;
                
                const now = Date.now();
                if (now - lastTime >= interval) {
                    if (!video.paused && !video.ended) {
                        try {
                            await pose.send({ image: video });
                            lastTime = now;
                        } catch (err) {
                            console.error('Frame error:', err);
                        }
                    }
                }
                
                if (!video.ended && isRunning) {
                    requestAnimationFrame(processFrame);
                } else {
                    isRunning = false;
                    startBtn.disabled = false;
                    statusDiv.innerHTML = '<i class="fas fa-check-circle text-green-500 mr-1"></i> Analisis selesai!';
                    progressBar.style.width = '100%';
                    progressText.textContent = '100%';
                    
                    console.log(`Total frame: ${poseData.length}`);
                    
                    // Analisis data
                    const result = analyzePoseData(poseData);
                    updateResults(result);
                    
                    if (result) {
                        console.log('Hasil analisis:', result);
                    }
                }
            }
            
            requestAnimationFrame(processFrame);
            
        } catch (error) {
            console.error('Error:', error);
            alert('Error: ' + error.message);
            statusDiv.innerHTML = '<i class="fas fa-exclamation-circle text-red-500 mr-1"></i> Error: ' + error.message;
            startBtn.disabled = false;
            isRunning = false;
        }
    }
